        """Summary of method

        Args:
            param1: The first parameter.
            param2: The second parameter.

        Returns:
            True if successful, False otherwise.

        """
        return ctx.message.author.id in self.bot.owner_ids

    """ Commands """

    @comms.command()
    async def test(self, ctx):

        Args:
            param1: The first parameter.
            param2: The second parameter.

        Returns:
            True if successful, False otherwise.

        """
        pass


def setup(bot):
    bot.add_cog(Cog_Type(bot))
